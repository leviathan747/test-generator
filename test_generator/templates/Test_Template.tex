% stem of a multipart question: typeset into a box now but held back, so
% the next \fitquestion (the first part) measures the stem and the part
% together and page-breaks before both if the pair doesn't fit
\newsavebox{\fitstembox}
\newif\ifpendingstem
\newcommand{\fitstem}[1]{%
  \sbox{\fitstembox}{%
    \setlength{\currentparskip}{\parskip}
    \begin{minipage}{\linewidth}
    \setlength{\parskip}{\currentparskip}
      #1
    \end{minipage}
  }%
  \global\pendingstemtrue
}

\newcommand{\fitquestion}[1]{%
  % typeset content into a box
  \sbox{\fitbox}{%
    \setlength{\currentparskip}{\parskip}
    \begin{minipage}{\linewidth}
    \setlength{\parskip}{\currentparskip}
      #1
    \end{minipage}
  }%

  % remaining space on page
  \dimen0=\dimexpr\pagegoal-\pagetotal\relax

  % full size of the box (the minipage centers on the baseline, so about
  % half of the content ends up in the depth) plus the glue added when
  % the box is placed on the page
  \dimen1=\dimexpr\ht\fitbox+\dp\fitbox+\parskip+\baselineskip\relax

  % a held-back stem must land on the same page as this box
  \ifpendingstem
    \advance\dimen1 \dimexpr\ht\fitstembox+\dp\fitstembox+\parskip+\baselineskip\relax
  \fi

  % if it doesn't fit, force page break
  \ifdim\dimen1>\dimen0
    \newpage
  \fi

  % print the held-back stem first, out-dented from the parts list back
  % to the full question width it was typeset at
  \ifpendingstem
    \par\noindent\hspace*{\dimexpr\linewidth-\wd\fitstembox\relax}%
    \usebox{\fitstembox}\par
    \global\pendingstemfalse
  \fi

  % print content
  \usebox{\fitbox}
}

% place a figure to the right of question content
% usage: \begin{figright}{<figure content>} <question text/choices> \end{figright}
% Falls back to full-width text with the figure centered below when the
% figure would leave less than ~12em of text column.
\newsavebox{\figrightbox}
\newlength{\figrightparskip}
\newif\iffigrightfits
\newenvironment{figright}[1]{%
  \sbox{\figrightbox}{#1}%
  \ifdim\wd\figrightbox<\dimexpr\linewidth-12em\relax
    \figrightfitstrue
  \else
    \figrightfitsfalse
  \fi
  \iffigrightfits
    \setlength{\figrightparskip}{\parskip}%
    \par\noindent
    \begin{minipage}[t]{\dimexpr\linewidth-\wd\figrightbox-1em\relax}%
      \setlength{\parskip}{\figrightparskip}%
  \fi
}{%
  \iffigrightfits
    \end{minipage}\hfill
    \begin{minipage}[t]{\wd\figrightbox}%
      \vspace{0pt}%
      \usebox{\figrightbox}%
    \end{minipage}\par
  \else
    \par\begin{center}\usebox{\figrightbox}\end{center}%
  \fi
}

% place a grading rubric to the right of solution content
% usage: \begin{gradedsolution}{<points & criterion \\ rows>} <solution> \end{gradedsolution}
% Falls back to full-width solution content with the rubric flush right
% below it when the rubric would leave less than ~12em of solution column.
\newsavebox{\rubricbox}
\newlength{\rubricparskip}
\newif\ifrubricfits
\newenvironment{gradedsolution}[1]{%
  \sbox{\rubricbox}{$\left\{\,\begin{tabular}{@{}r@{\quad}l@{}}#1\end{tabular}\right.$}%
  \ifdim\wd\rubricbox<\dimexpr\linewidth-12em\relax
    \rubricfitstrue
  \else
    \rubricfitsfalse
  \fi
  \ifrubricfits
    \setlength{\rubricparskip}{\parskip}%
    \par\noindent
    \begin{minipage}[t]{\dimexpr\linewidth-\wd\rubricbox-1em\relax}%
      \setlength{\parskip}{\rubricparskip}%
  \fi
}{%
  \ifrubricfits
    \end{minipage}\hfill
    \begin{minipage}[t]{\wd\rubricbox}%
      \vspace{0pt}%
      \usebox{\rubricbox}%
    \end{minipage}\par
  \else
    \par\hfill\usebox{\rubricbox}\par
  \fi
}
