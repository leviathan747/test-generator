% environment for placing a figure to the right with text on the left
% YAML questions omit \end{figureright}; Question_Template.tex closes it automatically
% usage: \begin{figureright}[width]{figure content}
\newenvironment{figureright}[2][3in]{%
  \def\figrightwidth{#1}%
  \def\figrightcontent{#2}%
  \global\figurerighttrue%
  \noindent\begin{minipage}[t]{\dimexpr\linewidth-#1-1em\relax}%
}{%
  \end{minipage}%
  \hfill%
  \begin{minipage}[t]{\figrightwidth}%
    \figrightcontent%
  \end{minipage}%
  \par%
  \global\figurerightfalse%
}
