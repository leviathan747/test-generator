\documentclass{standalone}
\usepackage{tikz}
\begin{document}
\begin{tikzpicture}[>=stealth, scale=0.75]
  % f(x) = 1/(x - 2): vertical asymptote at x = 2
  \begin{scope}
    \clip (-1.4,-3.8) rectangle (5.4,3.8);
    \draw[blue, thick, samples=120, domain=-1.4:1.74] plot (\x, {1/(\x-2)});
    \draw[blue, thick, samples=120, domain=2.26:5.4] plot (\x, {1/(\x-2)});
  \end{scope}
  % Vertical asymptote
  \draw[dashed] (2,-3.8) -- (2,3.8);
  % Axes
  \draw[->] (-1.6,0) -- (5.6,0) node[right] {$x$};
  \draw[->] (0,-4.0) -- (0,4.0) node[above] {$y$};
  % x-axis ticks
  \foreach \x in {-1,1,2,3,4,5} {
    \draw (\x,0.07) -- (\x,-0.07) node[below, font=\small] {\x};
  }
  % y-axis ticks
  \foreach \y in {-3,-2,-1,1,2,3} {
    \draw (0.07,\y) -- (-0.07,\y) node[left, font=\small] {\y};
  }
\end{tikzpicture}
\end{document}
