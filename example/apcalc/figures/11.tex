\documentclass{standalone}
\usepackage{tikz}
\begin{document}
\begin{tikzpicture}[>=stealth, xscale=1.4, yscale=0.8]
  % --- axes ---
  \draw[->] (-0.6,0) -- (5.8,0);          % x-axis
  \draw[->] (0,-1.2) -- (0,9.2);          % y-axis

  % x-axis ticks and labels
  \foreach \x in {1,2,3,4,5} {
    \draw (\x,0.08) -- (\x,-0.08);
    \node[anchor=north, font=\scriptsize] at (\x,-0.12) {\x};
  }
  % y-axis ticks and labels
  \foreach \y in {1,2,3,4,5,6,7,8} {
    \draw (0.08,\y) -- (-0.08,\y);
    \node[anchor=east, font=\scriptsize] at (-0.12,\y) {\y};
  }

  % --- curve g'(x): cubic, local max ~(1.3,5.67), local min ~(3.5,-1.32) ---
  % g'(x) = 3.944*(x^3/3 - 2.4*x^2 + 4.55*x) - 4.546  (extrema at x=1.3 and x=3.5; passes through (2,4))
  \begin{scope}
    \clip (-0.6,-1.5) rectangle (5.8,9.2);
    \draw[thick, domain=0.0:5.05, samples=220, smooth]
      plot (\x, {3.944*((\x*\x*\x)/3 - 2.4*\x*\x + 4.55*\x) - 4.546});
  \end{scope}

  % point (2,4) and label
  % node-based dot keeps a round shape under non-uniform axis scaling
  \node[circle, fill=black, inner sep=0pt, minimum size=6pt] at (2,4) {};
  \node[anchor=west] at (2.12,4.0) {$(2,\,4)$};

  % curve label g'
  \node at (5.2,7.6) {$g'$};
\end{tikzpicture}
\end{document}
