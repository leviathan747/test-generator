\documentclass{standalone}
\usepackage{tikz}
\begin{document}
\begin{tikzpicture}[>=stealth, scale=0.75]
  % Piecewise linear function on [-4, 4]:
  % removable discontinuity at x = -2 (limit 1, f(-2) = 2)
  % jump discontinuity at x = 1 (left limit 1, f(1) = 3 = right limit)
  \draw[blue, thick] (-4,-1) -- (-2,1);
  \draw[blue, thick] (-2,1) -- (1,1);
  \draw[blue, thick] (1,3) -- (4,0);
  % Axes
  \draw[->] (-4.6,0) -- (4.7,0) node[right] {$x$};
  \draw[->] (0,-1.8) -- (0,3.7) node[above] {$y$};
  % x-axis ticks
  \foreach \x in {-4,-3,-2,-1,1,2,3,4} {
    \draw (\x,0.07) -- (\x,-0.07) node[below, font=\small] {\x};
  }
  % y-axis ticks
  \foreach \y in {-1,1,2,3} {
    \draw (0.07,\y) -- (-0.07,\y) node[left, font=\small] {\y};
  }
  % Endpoints of the domain
  \fill[blue] (-4,-1) circle (3.5pt);
  \fill[blue] (4,0) circle (3.5pt);
  % Removable discontinuity at x = -2
  \draw[fill=white, draw=blue] (-2,1) circle (4pt);
  \fill[blue] (-2,2) circle (3.5pt);
  % Jump at x = 1: open circle from the left, f(1) = 3 from the right
  \draw[fill=white, draw=blue] (1,1) circle (4pt);
  \fill[blue] (1,3) circle (3.5pt);
\end{tikzpicture}
\end{document}
